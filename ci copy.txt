name: CI

on:
  push:
    branches: [develop]

jobs:
  install-dependencies-build-docker-image:
    if: ${{ github.ref != 'refs/heads/develop' && !contains(github.event.head_commit.message, '[skip ci]') }} # public; ignore commits with `[skip ci]`

    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
        with:
          # pulls all commits (needed for lerna to correctly version)
          # see https://stackoverflow.com/a/60184319/9285308 & https://github.com/actions/checkout
          fetch-depth: "0"

      - name: Pulls all tags (needed for lerna to correctly version)
        run: git fetch --depth=1 origin +refs/tags/*:refs/tags/* # see https://stackoverflow.com/a/60184319/9285308

      - name: Use NodeJS
        uses: actions/setup-node@v1
        with:
          node-version: "12.x"

      - name: Installing dependencies
        run: npm install

      - name: Lerna bootstrap
        uses: "MrBoolean/action-lerna@master"
        with:
          args: "bootstrap"

      - name: Formating code
        run: npm run format:check

      - name: Linter code
        run: npm run lint

      - name: Testing projects
        run: npm run test

      - run: git config --global user.email ${{ secrets.GIT_USER_EMAIL }}
      - run: git config --global user.name ${{ secrets.GIT_USER_NAME }}

      # SET GIT CREDENTIALS
      - name: Configure git credentials
        uses: OleksiyRudenko/gha-git-credentials@v1
        with:
          token: ${{ secrets.GITHUB_TOKEN }}

      # DOCKER HUB LOGIN
      - name: Login to docker hub
        if: success()
        uses: actions-hub/docker/login@master
        env:
          DOCKER_USERNAME: ${{ secrets.DOCKER_USERNAME }}
          DOCKER_PASSWORD: ${{ secrets.DOCKER_PASSWORD }}

      # Print what changed
      - run: ./node_modules/.bin/lerna changed

      - name: Create a new version
        run: yarn new-version

      - name: Get last changed tag
        run: |
          echo "::set-env name=LATEST_TAG::$(git describe --tags --abbrev=0)"

      - run: echo "$LATEST_TAG"

      - run: yarn ci:deploy
